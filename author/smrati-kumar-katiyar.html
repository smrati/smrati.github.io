def matrix_multiplication(m1, m2):
    matrix1_row = len(m1)
    matrix1_col = len(m1[0])
    matrix2_row = len(m2)
    matrix2_col = len(m2[0])

    if matrix1_col != matrix2_row:
        raise Exception("Matrix size not compatible for multiplication")

    final_matrix = []
    for i in range(matrix1_row):
        this_row = []
        for n in range(matrix2_col):
            element = 0
            for j in range(matrix1_col):
                element += (m1[i][j] * m2[j][n])
            this_row.append(element)
        final_matrix.append(this_row)
    return final_matrix
