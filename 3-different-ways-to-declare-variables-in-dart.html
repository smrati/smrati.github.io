we will learn about difference in these 3 types

mutable, static type

main() {
  mutableBasicType();
}

void mutableBasicType(){
  int a = 5;
  double b = 12.7;
  bool c = true;


  print(a);
  print(b);
  print(c);

}

A variable once assigned as int can't be reassigned as double

main() {
  mutableBasicType();
}

void mutableBasicType(){
  int a = 5;
  double b = 12.7;
  bool c = true;

  a = 45.6;


  print(a);
  print(b);
  print(c);

}

values of different type assigned to same variable as it was declared as dynamic

main() {
  mutableDynamic();
}

void mutableDynamic(){
  dynamic a = 14.6;
  print(a);
  a = false;
  print(a);
}

How to identify runtimetype

main() {
  mutableDynamic();
}

void mutableDynamic(){
  dynamic a = 14.6;
  print(a);
  print(a.runtimeType);
  a = false;
  print(a);
  print(a.runtimeType);
}
