Matrix Multiplication in Pytorch

import torch
x = torch.rand(1, 6400)
y = torch.rand(6400, 5000)

device = 'cuda' if torch.cuda.is_available() else 'cpu'

# register tensor object on available device
x, y = x.to(device), y.to(device)

z = x@y # Matrix Multiplication
